    25                                  %macro InterruptHandler 2
    26                                  	global %1	; advertise our function name as global symbol
    27                                  	extern %2	; declare C++ fcn of no args as accessible symbol
    28                                  %1:			; define our global symbol location
    29                                  	cli		; disable other interrupts for now
    30                                  	pushad		; save ALL registers (kind of extreme, but what the hell)
    31                                  	call %2		; call the C++ function with no arguments
    32                                  	mov al, 0x20	; load code to acknowledge interrupt
    33                                  	out 0x20, al	; whack 8259 PIC so it shuts up
    34                                  	popad		; restore ALL the registers
    35                                  	sti		; re-enable interrupts
    36                                  	iret		; return from exception
    37                                  %endmacro
