Introduction

  • Write a program to import a fixed length text file into a pre-existing SQL table.

Field classes

A field within a fixed length text file has a starting position (offset) and length (count). The field interface provides methods to get these properties. A field also responds to a read() event. Because its value can be converted into a number of different data types, each data type should be represented by a different Java class, which extends the base field class. 

public interface field {
  public String getName();
  public void setName( String v );
  public int getOffset();
  public int getCount();
  public void read( String v );
}

The base field implements the Field interface. 

public abstract basefield
    extends field {
  public basefield( int o, int l ) {
    setOffset( o );
    setLength( l );
  }
  public String getName() {
    return name;
  }
  public void setName( String v ) {
    name = v;
  }
  public int getOffset() {
    return offset;
  }
  public void setOffset( int v ) {
    offset = v;
  }
  public int getLength() {
    return length;
  }
  public void setLength( int v ) {
    length = v;
  }
  public void read( String v ) {
    if ( v == null ) {
      setValue( null );
      return;
    }
    setValue( v.substring( offset, length ) );
  }
  protected abstract void setValue( String v );
  private String name;
  private int offset;
  private int length;
}

Base record

A record maintains a list of fields. It implements a bind() method to bind custom fields to the record. 

public class record {
  public void bind( field f, String n ) {
    if ( f == null ) {
      return;
    }
    f.setName( n );
    if ( ptop == null ) {
      ptop = f;
      return;
    }
    field pcurr = ptop;
    for (;;) {
      if ( pcurr.getNext() == null ) {
        pcurr.setNext( f );
      }

      pcurr = pcurr.getNext();
    }
  }
  private field ptop;
}
