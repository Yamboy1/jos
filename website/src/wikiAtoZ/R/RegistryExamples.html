Example 1 - Creating a registry

Because it is an interface, a new instance of Registry cannot be created. A registry interface should be used throughout your code. An implementation of this interface must create a new registry. The following example demonstrates enough of the mechanism to load a registry.

  public void exampleA() {
	 Registry r = new BasicRegistry();
	 r.setString( "key1", "value1" );
	 r.setString( "key2", "value2" );
  }

  public void exampleB() {
	 Registry r = new BasicRegistry();
	 r.setString( "key1", "value1" );
	 r.setString( "key2", "value2" );
	 r.saveAs( "x-registry:file:example" );
  }

Example 2 - Loading a registry (direct)

It is possible to load a registry directly, without going throught a registry factory. The following example demonstrates enough of the mechanism to load a registry. Note: A registry factory itself must be configured. It loads its own configuration this way.

  public void example() {
	 Registry r = new BasicRegistry( "example" );
	 String v1 = r.getString( "key1" );
	 String v2 = r.getString( "key2" );
  }

  public void exampleSystem() {
	 Registry r = SystemRegistry.getRegistry();
	 String v1 = r.getString( "os.version" );
  }

  public void exampleNetwork() {
	 Registry r = NetworkRegistry.getRegistry();
	 String v1 keys = r.getString( "net.version" );
  }

Example 3 - Loading a registry (factory)

Use a registry factory to load a registry. A registry factory is strongly recommended. The following example uses the system-wide registry factory to load a registry.

  public void exampleNew() {
	 String url = "x-registry:ram:";
	 Registry r = RegistryFactory.getRegistry( url );
	 r.setString( "key1", "value1" );
	 r.setString( "key2", "value2" );
	 r.saveAs( "x-registry:file:example" );
  }

  public void exampleSystem() {
	 String url = "x-registry:os:";
	 Registry root = RegistryFactory.getRegistry( url );
	 Registry user = root.getRegistry( "user" );
	 Registry currentUser = user.getRegistry( "user1" );
	 Registry editor = currentUser.getRegistry( "editor" );
	 String s = r.getString( "most recently used file" );
  }

  public void exampleUserEditor() {
	 String url = "x-registry:os:#user/user1/editor";
	 Registry r = RegistryFactory.getRegistry( url );
	 String s = r.getString( "most recently used file" );
  }

  public void exampleNetwork() {
	 String url = "x-registry:net:";
	 Registry net = RegistryFactory.getRegistry( url );
	 Registry hosts = net.getRegistry( "hosts" );
	 Registry host = hosts.getRegistry( "host1" );
	 Registry user = root.getRegistry( "user" );
	 Registry currentUser = user.getRegistry( "user1" );
	 Registry editor = currentUser.getRegistry( "editor" );
	 String s = r.getString( "most recently used file" );
  }

  public void exampleNetworkUserEditor() {
	 String url = "x-registry:net:#host/host1/user/user1/editor";
	 Registry r = RegistryFactory.getRegistry( url );
	 String s = r.getString( "most recently used file" );
  }

  public void exampleFile() {
	 String url = "x-registry:file:example";
	 Registry r = RegistryFactory.getRegistry( url );
	 String v1 = r.getString( "key1" );
	 String v2 = r.getString( "key2" );
  }

  public void exampleHTTP() {
	 String url = "x-registry:http://host.domain/example";
	 Registry r = RegistryFactory.getRegistry( url );
	 String v1 = r.getString( "key1" );
	 String v2 = r.getString( "key2" );
  }

  public void exampleFTP() {
	 String url = "x-registry:ftp://host.domain/example";
	 Registry r = RegistryFactory.getRegistry( url );
	 String v1 = r.getString( "key1" );
	 String v2 = r.getString( "key2" );
  }

Example 4 - Accessing registry data

A registry supports all kinds of primative data in Java. The following example demonstrates each of the accessor methods.

  public void accessors() {
	 String url = "x-registry:file:accessors";
	 Registry r = RegistryFactory.getRegistry( url );
	 boolean f1 = r.getBoolean( "f1" );
	 byte f2 = r.getByte( "f2" );
	 char f3 = r.getChar( "f3" );
	 short f4 = r.getShort( "f4" );
	 int f5 = r.getInt( "f5" );
	 long f6 = r.getLong( "f6" );
	 float f7 = r.getFloat( "f7" );
	 double f8 = r.getDouble( "f8" );
	 String f9 = r.getString( "f9" );
	 Registry f10 = r.getRegistry( "f10" );

	 boolean[] f11 = r.getBooleanArray( "f11" );
	 byte[] f12 = r.getByteArray( "f12" );
	 char[] f13 = r.getCharArray( "f13" );
	 short[] f14 = r.getShortArray( "f14 );
	 int[] f15 = r.getIntArray( "f15 );
	 long[] f16 = r.getLongArray( "f16" );
	 float[] f17 = r.getFloatArray( "f17" );
	 double[] f18 = r.getDoubleArray( "f18" );
	 String[] f19 = r.getStringArray( "f19" );
	 Registry[] f20 = r.getRegistryArray( "f20" );
  }

Example 5 - Modifying registry data

A registry supports all kinds of primative data in Java. The following example demonstrates each of the modifier methods.

  public void modifiers() {
	 String url = "x-registry:file:modifiers";
	 Registry r = RegistryFactory.getRegistry( url );
	 r.setBoolean( "f1", true );
	 r.setByte( "f2", (byte) 0x00 );
	 r.setChar( "f3", (char) 0x00 );
	 r.setShort( "f4", (short) 0x00 );
	 r.setInt( "f5", 0x00 );
	 r.setLong( "f6", (long) 0x00 );
	 r.setFloat( "f7", Float.intBitsToFloat( 0x00 ) );
	 r.setDouble( "f8", Double.longBitsToDouble( (long) 0x00 ) );
	 r.setString( "f9", "value9" );
	 r.setRegistry( "f10", RegistryFactory.getRegistry( url ) );

	 r.setBooleanArray( "f11", createBooleanArray() );
	 r.setByteArray( "f12", createByteArray() );
	 r.setCharArray( "f13", "value13".toCharArray() );
	 r.setShortArray( "f14, createShortArray() );
	 r.setIntArray( "f15", createIntArray() );
	 r.setLongArray( "f16", createLongArray() );
	 r.setFloatArray( "f17", createFloatArray() );
	 r.setDoubleArray( "f18", createDoubleArray() );
	 r.setStringArray( "f19", createStringArray() );
	 r.setRegistryArray( "f20", createRegistryArray() );
  }
